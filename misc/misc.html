Misc / Gradient Text

Make your text stand out with a gradient. This is done by using the background-clip CSS property.

Gradient Text - woah, so cool and fancy!

.gradient-text {
  background: -webkit-linear-gradient(0, #c9ff05, #7a7aff);
  -webkit-background-clip: text;
  -webkit-text-fill-color: transparent;
}

Misc / Random String Generator

Generate a random string of characters.

function generateRandomString(length) {
	const chars = '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ';
	let result = '';
	for (let i = length; i > 0; --i) result += chars[Math.floor(Math.random() * chars.length)];
	return result;
}

document.getElementById('generate').addEventListener('click', function () {
	document.getElementById('random-string').value = generateRandomString(document.getElementById('random-string-length').value);
});
